5. **Prepare Linux Assets step** -- Add after "Prepare Windows Assets":

```yaml
- name: Prepare Linux Assets
  if: matrix.os == 'ubuntu-22.04'
  shell: bash
  run: |
    set -euo pipefail

    mkdir -p release-assets
    VERSION="${GITHUB_REF_NAME}"
    ARCH="${{ matrix.arch }}"
    TARGET="${{ matrix.target }}"

    BUNDLE_DIRS=(
      "src-tauri/target/${TARGET}/release/bundle"
      "src-tauri/target/release/bundle"
    )

    TAR_GZ=""
    DEB=""
    APPIMAGE=""
    for dir in "${BUNDLE_DIRS[@]}"; do
      if [ -d "${dir}" ]; then
        if [ -z "${TAR_GZ}" ]; then
          TAR_GZ="$(find "${dir}" -type f -name "*.tar.gz" | head -n 1 || true)"
        fi
        if [ -z "${DEB}" ]; then
          DEB="$(find "${dir}" -type f -name "*.deb" | head -n 1 || true)"
        fi
        if [ -z "${APPIMAGE}" ]; then
          APPIMAGE="$(find "${dir}" -type f -name "*.AppImage" | head -n 1 || true)"
        fi
      fi
    done

    if [ -z "${TAR_GZ}" ]; then
      echo "ERROR: No *.tar.gz updater artifact found (target=${TARGET}, arch=${ARCH})." >&2
      for dir in "${BUNDLE_DIRS[@]}"; do
        echo "---- list: ${dir}" >&2
        ls -la "${dir}" 2>/dev/null || true
        find "${dir}" -maxdepth 6 -type f 2>/dev/null | head -n 200 >&2 || true
      done
      exit 1
    fi

    if [ ! -f "${TAR_GZ}.sig" ]; then
      echo "ERROR: No updater signature file: ${TAR_GZ}.sig" >&2
      exit 1
    fi

    NEW_TAR_GZ="Skills-Hub-${VERSION}-Linux-${ARCH}.tar.gz"
    cp "${TAR_GZ}" "release-assets/${NEW_TAR_GZ}"
    cp "${TAR_GZ}.sig" "release-assets/${NEW_TAR_GZ}.sig"

    if [ -n "${DEB}" ]; then
      NEW_DEB="Skills-Hub-${VERSION}-Linux-${ARCH}.deb"
      cp "${DEB}" "release-assets/${NEW_DEB}"
    fi

    if [ -n "${APPIMAGE}" ]; then
      NEW_APPIMAGE="Skills-Hub-${VERSION}-Linux-${ARCH}.AppImage"
      cp "${APPIMAGE}" "release-assets/${NEW_APPIMAGE}"
    fi
```
