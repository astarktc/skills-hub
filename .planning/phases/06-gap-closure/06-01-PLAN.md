```rust
pub fn list_assignments_with_staleness(
    store: &SkillStore,
    project_id: &str,
) -> Result<Vec<ProjectSkillAssignmentRecord>> {
    let assignments = store.list_project_skill_assignments(project_id)?;
    let mut result = Vec::with_capacity(assignments.len());

    for mut assignment in assignments {
        // D-05: Check for missing source or target before staleness
        let skill_opt = store.get_skill_by_id(&assignment.skill_id).ok().flatten();
        let source_exists = skill_opt
            .as_ref()
            .map(|s| Path::new(&s.central_path).exists())
            .unwrap_or(false);

        // Resolve target path for existence check
        let target_exists = if let Some(ref skill) = skill_opt {
            if let Some(adapter) = tool_adapters::adapter_by_key(&assignment.tool) {
                let project = store.get_project_by_id(&assignment.project_id).ok().flatten();
                project.map(|p| {
                    let target = resolve_project_sync_target(
                        Path::new(&p.path),
                        adapter.relative_skills_dir,
                        &skill.name,
                    );
                    target.exists() || target.symlink_metadata().is_ok()
                }).unwrap_or(false)
            } else {
                false
            }
        } else {
            false
        };

        if !source_exists || (!target_exists && (assignment.status == "synced" || assignment.status == "stale")) {
            // D-04/D-06: Source absent or target absent -> missing, persist to DB
            if assignment.status != "missing" {
                let _ = store.update_assignment_status(
                    &assignment.id,
                    "missing",
                    None,
                    None,
                    None,
                    None,
                );
            }
            assignment.status = "missing".to_string();
        } else if assignment.status == "synced" && assignment.mode == "copy" {
            // D-09: Original staleness check for copy-mode synced targets
            if let Some(ref stored_hash) = assignment.content_hash {
                if let Some(ref skill) = skill_opt {
                    let source = Path::new(&skill.central_path);
                    if source.exists() {
                        if let Ok(current_hash) = content_hash::hash_dir(source) {
                            if current_hash != *stored_hash {
                                assignment.status = "stale".to_string();
                                let _ = store.update_assignment_status(
                                    &assignment.id,
                                    "stale",
                                    None,
                                    None,
                                    None,
                                    None,
                                );
                            }
                        }
                    }
                }
            }
        }
        // D-07: If source reappears and target exists, the above conditions
        // will not trigger missing, so status stays at whatever it was
        // (or gets recalculated normally via staleness check)

        result.push(assignment);
    }

    Ok(result)
}
```

- Moved skill lookup BEFORE the staleness check (was inside the copy-mode block)
- Added source_exists check: if skill record missing or central_path doesn't exist -> missing
- Added target_exists check: resolve target path, check exists or symlink_metadata
- Target-absent only triggers missing for "synced" or "stale" statuses (not "pending" or "error" -- those already indicate problems)
- Missing detection runs BEFORE the copy-mode staleness check
- Staleness check preserved as-is for the non-missing case
