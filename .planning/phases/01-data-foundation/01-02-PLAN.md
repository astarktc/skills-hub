**register_project_path** -- validates, canonicalizes, checks duplicates, creates record. Note: `expand_home_path` is passed as a callback so the core layer does not depend directly on the commands layer. The caller (commands/projects.rs) provides `crate::commands::expand_home_path`:

```rust
use std::path::PathBuf;

pub fn register_project_path(
    store: &SkillStore,
    path: &str,
    now_ms: i64,
    expand_home: impl Fn(&str) -> Result<PathBuf>,
) -> Result<ProjectDto> {
    // D-08: expand ~ and canonicalize using the shared helper
    let expanded = expand_home(path)?;
    let canonical = std::fs::canonicalize(&expanded)
        .with_context(|| format!("failed to resolve path: {:?}", expanded))?;

    // D-09: validate directory exists
    if !canonical.is_dir() {
        bail!("path is not a directory: {:?}", canonical);
    }

    let path_str = canonical.to_string_lossy().to_string();

    // D-09: check for duplicate registration
    if store.get_project_by_path(&path_str)?.is_some() {
        bail!("project already registered: {}", path_str);
    }

    let record = ProjectRecord {
        id: Uuid::new_v4().to_string(),
        path: path_str,
        created_at: now_ms,
        updated_at: now_ms,
    };
    store.register_project(&record)?;
    to_project_dto(&record, store)
}
```

**remove_project_with_cleanup** -- reads assignments, removes filesystem artifacts using sync_engine::remove_path_any, cascade-deletes. Per D-07, this reuses the existing sync engine primitive rather than creating a duplicate:

```rust
pub fn remove_project_with_cleanup(store: &SkillStore, project_id: &str) -> Result<()> {
    // D-07: read assignment paths before DB delete
    let project = store.get_project_by_id(project_id)?
        .ok_or_else(|| anyhow::anyhow!("project not found: {}", project_id))?;

    let assignments = store.list_project_skill_assignments(project_id)?;

    // For each synced/stale assignment, compute target path and remove filesystem artifact
    // using existing sync_engine::remove_path_any per D-07
    for assignment in &assignments {
        if assignment.status == "synced" || assignment.status == "stale" {
            if let Ok(Some(skill)) = store.get_skill_by_id(&assignment.skill_id) {
                if let Some(adapter) = tool_adapters::adapter_by_key(&assignment.tool) {
                    let project_path = Path::new(&project.path);
                    let target = project_path
                        .join(adapter.relative_skills_dir)
                        .join(&skill.name);
                    // Best-effort removal using existing sync_engine primitive
                    // remove_path_any handles symlinks, dirs, files, and NotFound gracefully
                    if let Err(e) = sync_engine::remove_path_any(&target) {
                        log::warn!("failed to remove {:?}: {}", target, e);
                    }
                }
            }
        }
    }

    // CASCADE delete handles project_tools and project_skill_assignments rows
    store.delete_project(project_id)?;
    Ok(())
}
```

pub async fn register_project(
    store: State<'_, SkillStore>,
    path: String,

    tauri::async_runtime::spawn_blocking(move || {
