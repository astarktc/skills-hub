```rust
#[tauri::command]
pub async fn register_project(
    store: State<'_, SkillStore>,
    path: String,
) -> Result<ProjectDto, String> {
    let store = store.inner().clone();
    tauri::async_runtime::spawn_blocking(move || {
        // D-08: expand ~ and canonicalize
        let expanded = expand_home_path(&path)?;
        let canonical = std::fs::canonicalize(&expanded)
            .with_context(|| format!("failed to resolve path: {:?}", expanded))?;

        // D-09: validate directory exists
        if !canonical.is_dir() {
            anyhow::bail!("path is not a directory: {:?}", canonical);
        }

        let path_str = canonical.to_string_lossy().to_string();

        // D-09: check for duplicate registration
        if store.get_project_by_path(&path_str)?.is_some() {
            anyhow::bail!("project already registered: {}", path_str);
        }

        let now = now_ms();
        let record = ProjectRecord {
            id: Uuid::new_v4().to_string(),
            path: path_str,
            created_at: now,
            updated_at: now,
        };
        store.register_project(&record)?;
        to_project_dto(&record, &store)
    })
    .await
    .map_err(|e| e.to_string())?
    .map_err(format_anyhow_error)
}
```

**remove_project** (per D-07: reads assignment paths, removes filesystem artifacts, then cascade-deletes):

```rust
#[tauri::command]
pub async fn remove_project(
    store: State<'_, SkillStore>,
    projectId: String,
) -> Result<(), String> {
    let store = store.inner().clone();
    tauri::async_runtime::spawn_blocking(move || {
        // D-07: read assignment paths before DB delete
        let project = store.get_project_by_id(&projectId)?
            .ok_or_else(|| anyhow::anyhow!("project not found: {}", projectId))?;

        let assignments = store.list_project_skill_assignments(&projectId)?;

        // For each synced assignment, compute target path and remove filesystem artifact
        // Target path = project.path / tool_adapter.relative_skills_dir / skill_name
        // But we need skill_name from skills table -- join via skill_id
        for assignment in &assignments {
            if assignment.status == "synced" || assignment.status == "stale" {
                // Look up the skill to get its name for path computation
                if let Ok(Some(skill)) = store.get_skill_by_id(&assignment.skill_id) {
                    if let Some(adapter) = crate::core::tool_adapters::adapter_by_key(&assignment.tool) {
                        let project_path = std::path::Path::new(&project.path);
                        let target = project_path
                            .join(adapter.relative_skills_dir)
                            .join(&skill.name);
                        let target_str = target.to_string_lossy().to_string();
                        // Best-effort removal -- log but don't fail the whole delete
                        if let Err(e) = remove_path_any(&target_str) {
                            log::warn!("failed to remove {:?}: {}", target_str, e);
                        }
                    }
                }
            }
        }

        // CASCADE delete handles project_tools and project_skill_assignments rows
        store.delete_project(&projectId)?;
        Ok::<_, anyhow::Error>(())
    })
    .await
    .map_err(|e| e.to_string())?
    .map_err(format_anyhow_error)
}
```

NOTE: `remove_project` uses `projectId` (camelCase) as the parameter name per Tauri convention. Use `#[allow(non_snake_case)]` on the function if clippy complains.
