pub async fn remove_project(store: State<'_, SkillStore>, projectId: String) -> Result<(), String> {
    let store = store.inner().clone();
    tauri::async_runtime::spawn_blocking(move || {
        project_ops::remove_project_with_cleanup(&store, &projectId)
    })
    .await
    .map_err(|e| e.to_string())?
    .map_err(format_anyhow_error)
}

```rust
#[tauri::command]
#[allow(non_snake_case)]
pub async fn update_project_gitignore(
    store: State<'_, SkillStore>,
    projectId: String,
    addToGitignore: bool,
    addToExclude: bool,
) -> Result<(), String> {
    let store = store.inner().clone();
    tauri::async_runtime::spawn_blocking(move || {
        use std::fs;
        use std::path::Path;

        let project = store
            .get_project_by_id(&projectId)?
            .ok_or_else(|| anyhow::anyhow!("project not found: {}", projectId))?;

        let project_path = Path::new(&project.path);
        if !project_path.is_dir() {
            anyhow::bail!("project directory does not exist: {}", project.path);
        }

        // Collect unique relative_skills_dir patterns from the project's configured tools.
        // Each tool adapter has a relative_skills_dir (e.g. ".claude/skills", ".cursor/skills").
        // We collect unique directories and add them as gitignore patterns.
        let tools = store.list_project_tools(&projectId)?;
        let mut patterns: Vec<String> = Vec::new();
        for tool_record in &tools {
            if let Some(adapter) = crate::core::tool_adapters::adapter_by_key(&tool_record.tool) {
                let dir = adapter.relative_skills_dir;
                // Use the top-level directory component (e.g. ".claude" from ".claude/skills")
                // as the gitignore pattern, with trailing slash.
                let pattern = format!("/{}/", dir);
                if !patterns.contains(&pattern) {
                    patterns.push(pattern);
                }
            }
        }

        // If no tools configured yet, nothing to add
        if patterns.is_empty() {
            return Ok(());
        }

        let gitignore_block = format!(
            "\n# Skills Hub — managed skill directories\n{}\n",
            patterns.join("\n")
        );

        if addToGitignore {
            let gitignore_path = project_path.join(".gitignore");
            let existing = if gitignore_path.exists() {
                fs::read_to_string(&gitignore_path)
                    .with_context(|| format!("failed to read {}", gitignore_path.display()))?
            } else {
                String::new()
            };

            // Only append if the marker comment is not already present
            if !existing.contains("# Skills Hub") {
                let mut content = existing;
                if !content.ends_with('\n') && !content.is_empty() {
                    content.push('\n');
                }
                content.push_str(&gitignore_block);
                fs::write(&gitignore_path, content)
                    .with_context(|| format!("failed to write {}", gitignore_path.display()))?;
            }
        }

        if addToExclude {
            let exclude_path = project_path.join(".git").join("info").join("exclude");
            // .git/info/ directory may not exist in fresh repos
            if let Some(parent) = exclude_path.parent() {
                if !parent.exists() {
                    fs::create_dir_all(parent)
                        .with_context(|| format!("failed to create {}", parent.display()))?;
                }
            }

            let existing = if exclude_path.exists() {
                fs::read_to_string(&exclude_path)
                    .with_context(|| format!("failed to read {}", exclude_path.display()))?
            } else {
                String::new()
            };

            // Only append if the marker comment is not already present
            if !existing.contains("# Skills Hub") {
                let mut content = existing;
                if !content.ends_with('\n') && !content.is_empty() {
                    content.push('\n');
                }
                content.push_str(&gitignore_block);
                fs::write(&exclude_path, content)
                    .with_context(|| format!("failed to write {}", exclude_path.display()))?;
            }
        }

        Ok(())
    })
    .await
    .map_err(|e| e.to_string())?
    .map_err(format_anyhow_error)
}
```

**Key implementation details:**

- Uses `anyhow::Context` (already imported in the commands module) for error context on filesystem operations.
- Looks up the project by ID to get its path, then reads its configured tools to determine which `relative_skills_dir` patterns to add.
- Each pattern is formatted as `/<relative_skills_dir>/` (e.g. `/.claude/skills/`) -- the leading slash anchors it to the project root.
- Includes a `# Skills Hub` marker comment to prevent duplicate entries on re-registration.
- Creates `.git/info/` directory if it does not exist (handles fresh repos).
- Idempotent: if the marker comment already exists, no changes are made.
